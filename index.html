function startWizard() { const fullName = document.getElementById('instructorName').value.trim(); const cohort = document.getElementById('courseCohort').value.trim(); if (!fullName || !cohort) { alert("Please fill in both your name and cohort to continue."); return; } const firstName = fullName.split(' ')[0]; const safeName = fullName.replace(/\s+/g, '-').toLowerCase(); const safeCohort = cohort.replace(/\s+/g, '-').toLowerCase(); const projectName = `pm-lab-${safeName}-${safeCohort}`; // הלינק המתוקן שכולל גם את המשתנה וגם את השם המותאם const baseUrl = "https://vercel.com/new/clone?repository-url=https%3A%2F%2Fgithub.com%2Fmedusajs%2Fmedusa-starter-default"; const envVar = "env=MEDUSA_ADMIN_ONBOARDING_TYPE&envDescription=Select+the+onboarding+type&envLink=https%3A%2F%2Fdocs.medusajs.com"; // כאן אנחנו מחברים הכל יחד - שים לב לסדר הפרמטרים document.getElementById('deployLink').href = `${baseUrl}&${envVar}&project-name=${projectName}&repository-name=${projectName}&envValue=nextjs`; // עדכון הברכות document.getElementById('greeting1').innerText = `Step 1: Launch ${firstName}'s Store`; document.getElementById('greeting2').innerText = `Step 2: Access Your Data, ${firstName}`; document.getElementById('greeting3').innerText = `${firstName}, Setup Claude`; document.getElementById('greeting4').innerText = `Success, ${firstName}!`; document.getElementById('step0').classList.remove('active'); document.getElementById('step1').classList.add('active'); document.getElementById('progressBar').style.display = 'flex'; document.getElementById('navBtns').style.display = 'flex'; currentStep = 1; }
